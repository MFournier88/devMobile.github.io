Orientation

D'abords il vous faut installer la librairie expo-screen-orientation => npm install expo-screen-orientation

Ensuite modifier le fichier app.json pour que les deux propiétées orientation et infoPlist soit comme dans cet exemple :


{
  "expo": {
    "name": "app1",
    "slug": "app1",
    "scheme": "app1",
    "version": "1.0.0",
    "orientation": "default",
    "icon": "./assets/icon.png",
    "userInterfaceStyle": "light",
    "splash": {
      "image": "./assets/splash.png",
      "resizeMode": "contain",
      "backgroundColor": "#ffffff"
    },
    "ios": {
      "supportsTablet": true,
      "infoPlist": {
        "UISupportedInterfaceOrientations": [
          "UIInterfaceOrientationPortrait",
          "UIInterfaceOrientationLandscapeLeft",
          "UIInterfaceOrientationLandscapeRight"
        ]
      }
    },
    "android": {
      "adaptiveIcon": {
        "foregroundImage": "./assets/adaptive-icon.png",
        "backgroundColor": "#ffffff"
      },
       "orientation": "default"
    },
    "web": {
      "favicon": "./assets/favicon.png"
    },
    "plugins": [
      "expo-router"
    ]
  }
 }

Maintenant vous devez faire attention pour que votre affichage soit décent dans toutes les orientations.
